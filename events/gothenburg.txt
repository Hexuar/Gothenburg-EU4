country_event = {
    id = gothenburg.4
	title = gothenburg.fate_of_the_hansa_title
	desc = gothenburg.fate_of_the_hansa_desc
	picture = DIPLOMACY_eventPicture

	is_triggered_only = yes

	option = {
	    name = gothenburg.fate_of_the_hansa_take_control

		add_treasury = -1000

		if = {
		    limit = {
				HSA = { exists = yes }
			}
			create_subject = {
                subject_type = gothenburg.trade_city
                subject = HSA
            }
		}

		if = {
		    limit = {
				HAM = { exists = yes }
			}
            create_subject = {
                subject_type = gothenburg.trade_city
                subject = HAM
            }
        }

        if = {
		    limit = {
				BRE = { exists = yes }
			}
            create_subject = {
                subject_type = gothenburg.trade_city
                subject = BRE
            }
        }

        if = {
		    limit = {
				RIG = { exists = yes }
			}
            create_subject = {
                subject_type = gothenburg.trade_city
                subject = RIG
            }
        }

        hidden_effect = {
            HSA = {
                change_government = republic
                add_government_reform = gothenburg.scandinavian_trade_city
            }
            HAM = {
                change_government = republic
                add_government_reform = gothenburg.scandinavian_trade_city
            }
            BRE = {
                change_government = republic
                add_government_reform = gothenburg.scandinavian_trade_city
            }
            RIG = {
                change_government = republic
                add_government_reform = gothenburg.scandinavian_trade_city
            }
        }
	}

	option = {
	    name = gothenburg.fate_of_the_hansa_cooperate

		add_treasury = 500

		add_country_modifier = {
		    name = gothenburg.hansa_cooperation
			duration = 9125
		}
	}
}
