country_event = {
    id = gothenburg.2
	title = gothenburg.city_grows_title
	desc = gothenburg.city_grows_desc
	picture = CITY_VIEW_eventPicture

	trigger = {
	    tag = GBG
	}

	mean_time_to_happen = {
	    months = 18

		modifier = {
		    factor = 0.33
			4942 = {
      		    NOT = { development = 20 }
            }
		}
	}

	option = {
	    name = gothenburg.city_grows_accept

		if = {
		    limit = {
    			4942 = {
          		    NOT = { development = 20 }
                }
			}
			4942 = {
    		    add_base_tax = 1
    			add_base_production = 1
    		}
		}

		add_owned_provinces_development_ducats = {
		    multiplier = 2
			age_multiplier = 1
			trigger = { always = yes }
		}
	}
}
