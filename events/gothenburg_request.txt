namespace = gothenburg_request_province

country_event = {
	id = gothenburg_request_province.1
	title = gothenburg.request_aland_event_title
	desc = gothenburg.request_aland_event_desc
	picture = TRADEGOODS_eventPicture

	is_triggered_only = yes

	option = {
		name = gothenburg.request_aland_event.accept

		if = {
		    limit = {
				owns = 1930
				exists = FROM
			}

			1930 = {
			    cede_province = FROM
				add_core = FROM
				remove_core = ROOT
			}

			add_treasury = 100
            FROM = {
 			    add_treasury = -100
 			}
		}

		ai_chance = { factor = 90 }
	}
	option = {
	    name = gothenburg.request_aland_event.reject

		add_prestige = 10

		FROM = {
		    clr_country_flag = gothenburg.requested_bornholm
		}

		ai_chance = { factor = 10 }
	}
}
